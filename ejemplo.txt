Program TripletFactorial;

uses hola;

var
  n: integer;

function tripletFactorial(n: integer): integer;
begin
  if n <= 0 then
  begin
    tripletFactorial := 1;  // Caso base: si n es menor o igual a 0, devolvemos 1 
  end
  else
  begin
    tripletFactorial := n * tripletFactorial(n - 3);  // Llamada recursiva con n-3 
  end;
end;
